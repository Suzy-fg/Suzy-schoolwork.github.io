    <script>
        const apiKey = "";
        const canvas = document.getElementById('imageCanvas');
        const ctx = canvas.getContext('2d', { willReadFrequently: true });
        const fileInput = document.getElementById('fileInput');
        const wrapper = document.getElementById('canvasWrapper');

        let appState = {
            currentImg: null,
            filter: 'none',
            brightness: 1.0,
            contrast: 1.0,
            saturation: 1.0,
            isBrushActive: false,
            sparkles: []
        };

        function render() {
            if (!appState.currentImg) return;
            
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            // 基礎調色
            ctx.filter = `brightness(${appState.brightness}) contrast(${appState.contrast}) saturate(${appState.saturation})`;
            ctx.drawImage(appState.currentImg, 0, 0, canvas.width, canvas.height);
            ctx.filter = 'none';

            // 套用藝術氛圍濾鏡 (ImageData 處理，確保與調色疊加)
            if (appState.filter !== 'none') {
                const imgData = ctx.getImageData(0, 0, canvas.width, canvas.height);
                const d = imgData.data;
                for (let i = 0; i < d.length; i += 4) {
                    if (appState.filter === 'peach') { d[i] += 25; d[i+1] += 5; }
                    else if (appState.filter === 'mist') { d[i] += 15; d[i+2] += 25; }
                    else if (appState.filter === 'ocean') { d[i] *= 0.85; d[i+2] += 35; }
                    else if (appState.filter === 'noir') {
                        const g = d[i]*0.3 + d[i+1]*0.59 + d[i+2]*0.11;
                        d[i] = d[i+1] = d[i+2] = g;
                    } else if (appState.filter === 'golden') {
                        d[i] += 35; d[i+1] += 20; d[i+2] -= 10;
                    }
                }
                ctx.putImageData(imgData, 0, 0);
            }

            // 星塵繪製
            appState.sparkles.forEach(p => {
                const grad = ctx.createRadialGradient(p.x, p.y, 0, p.x, p.y, p.s);
                grad.addColorStop(0, 'rgba(255,255,255,1)');
                grad.addColorStop(0.5, 'rgba(255,255,255,0.4)');
                grad.addColorStop(1, 'transparent');
                ctx.fillStyle = grad;
                ctx.beginPath(); ctx.arc(p.x, p.y, p.s, 0, Math.PI*2); ctx.fill();
            });
        }

        // --- 檔案上傳 ---
        fileInput.onchange = e => {
            const reader = new FileReader();
            reader.onload = ev => {
                const img = new Image();
                img.onload = () => {
                    appState.currentImg = img;
                    document.getElementById('emptyState').style.display = 'none';
                    canvas.classList.remove('checkerboard');
                    setupCanvas();
                    render();
                };
                img.src = ev.target.result;
            };
            reader.readAsDataURL(e.target.files[0]);
        };

        function setupCanvas() {
            const img = appState.currentImg;
            const ratio = Math.min((window.innerWidth - 650) / img.width, (window.innerHeight - 250) / img.height, 1);
            canvas.width = img.width * ratio;
            canvas.height = img.height * ratio;
        }

        // --- 濾鏡功能 ---
        document.querySelectorAll('.filter-item').forEach(btn => {
            btn.onclick = () => {
                document.querySelectorAll('.filter-item').forEach(b => b.classList.remove('active'));
                btn.classList.add('active');
                appState.filter = btn.dataset.filter;
                render();
            };
        });

        // --- 調色控制 ---
        document.getElementById('adjBright').oninput = e => {
            appState.brightness = e.target.value / 100;
            document.getElementById('valBright').innerText = e.target.value + '%';
            render();
        };
        document.getElementById('adjContrast').oninput = e => {
            appState.contrast = e.target.value / 100;
            document.getElementById('valContrast').innerText = e.target.value + '%';
            render();
        };
        document.getElementById('adjSaturate').oninput = e => {
            appState.saturation = e.target.value / 100;
            document.getElementById('valSaturate').innerText = e.target.value + '%';
            render();
        };

        // --- AI 去背功能 ---
        document.getElementById('btnAiRemoveBg').onclick = async () => {
            if (!appState.currentImg) return;
            const btn = document.getElementById('btnAiRemoveBg');
            btn.classList.add('ai-loading');
            btn.innerText = "✨ AI 去背處理中...";

            const base64 = canvas.toDataURL('image/png').split(',')[1];
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-image-preview:generateContent?key=${apiKey}`;
            const payload = {
                contents: [{
                    parts: [
                        { text: "Remove the background of this image. Keep the main subject and make everything else transparent." },
                        { inlineData: { mimeType: "image/png", data: base64 } }
                    ]
                }],
                generationConfig: { responseModalities: ["IMAGE"] }
            };

            try {
                const res = await fetch(url, { method: 'POST', body: JSON.stringify(payload) });
                const result = await res.json();
                const newBase = result.candidates?.[0]?.content?.parts?.find(p => p.inlineData)?.inlineData?.data;
                if (newBase) {
                    const newImg = new Image();
                    newImg.onload = () => {
                        appState.currentImg = newImg;
                        canvas.classList.add('checkerboard');
                        render();
                    };
                    newImg.src = `data:image/png;base64,${newBase}`;
                }
            } catch (e) { alert("AI 去背目前連線繁忙，請稍後再試。"); }

            btn.classList.remove('ai-loading');
            btn.innerText = "✨ AI 一鍵自動去背景";
        };

        // --- AI 文字功能 ---
        async function callGemini(prompt) {
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
            const base64 = canvas.toDataURL('image/png').split(',')[1];
            const payload = { contents: [{ parts: [{ text: prompt }, { inlineData: { mimeType: 'image/png', data: base64 } }] }] };
            const res = await fetch(url, { method: 'POST', body: JSON.stringify(payload) });
            return await res.json();
        }

        document.getElementById('btnAiSuggest').onclick = async () => {
            if (!appState.currentImg) return;
            const btn = document.getElementById('btnAiSuggest');
            btn.classList.add('ai-loading');
            const result = await callGemini("這張照片適合什麼濾鏡與調色參數？請簡短說明理由。");
            if (result) alert(`✨ AI 專家建議：\n${result.candidates[0].content.parts[0].text}`);
            btn.classList.remove('ai-loading');
        };

        document.getElementById('btnAiCaption').onclick = async () => {
            if (!appState.currentImg) return;
            const btn = document.getElementById('btnAiCaption');
            btn.classList.add('ai-loading');
            const result = await callGemini("請為這張照片寫一段充滿少女心、優雅的 IG 文案，包含繁體中文標籤。");
            if (result) {
                document.getElementById('aiCaptionText').innerText = result.candidates[0].content.parts[0].text;
                document.getElementById('aiCaptionBox').classList.remove('hidden');
            }
            btn.classList.remove('ai-loading');
        };

        // --- 互動功能 ---
        document.querySelectorAll('.frame-btn').forEach(btn => {
            btn.onclick = () => {
                document.querySelectorAll('.frame-btn').forEach(b => b.classList.remove('active'));
                btn.classList.add('active');
                wrapper.className = btn.dataset.frame !== 'none' ? 'frame-' + btn.dataset.frame : '';
            };
        });

        document.getElementById('toolBrush').onclick = () => {
            appState.isBrushActive = !appState.isBrushActive;
            document.getElementById('toolBrush').classList.toggle('active');
            document.getElementById('toolBrush').innerText = appState.isBrushActive ? "🪄 星塵魔法施展中..." : "🪄 開啟仙女棒星塵筆刷";
        };

        canvas.onmousemove = e => {
            if (!appState.isBrushActive || !appState.currentImg) return;
            const rect = canvas.getBoundingClientRect();
            const x = (e.clientX - rect.left) * (canvas.width / rect.width);
            const y = (e.clientY - rect.top) * (canvas.height / rect.height);
            if (Math.random() > 0.6) {
                appState.sparkles.push({ x, y, s: Math.random() * 8 + 3 });
                render();
            }
        };

        document.getElementById('btnSave').onclick = () => {
            if (!appState.currentImg) return;
            const link = document.createElement('a');
            link.download = 'stardust-ultimate.png';
            link.href = canvas.toDataURL('image/png');
            link.click();
        };
    </script>
